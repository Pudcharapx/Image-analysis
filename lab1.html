Lab 1: Image Basics and ROI Extraction¶

ปฏิบัติการนี้มุ่งเน้นความเข้าใจพื้นฐานเรื่องรูปภาพ (Image) การเข้าถึงค่าพิกเซลด้วยตนเอง การใช้สูตรเปลี่ยนสีเป็น Grayscale และการตัดภาพบริเวณที่สนใจ (ROI)

# โจทย์ข้อที่ 1
# การเขียนตัวแปลงสีเทาด้วยสูตรคณิตศาสตร์แบบแมนนวล (Manual Grayscale Converter)

# import library ที่จำเป็นต่อการใช้งาน
import cv2 # นำเข้าไลบรารี OpenCV สำหรับการประมวลผลและอ่านไฟล์ภาพ
import numpy as np # นำเข้าไลบรารี NumPy สำหรับจัดการอาร์เรย์และคำนวณทางคณิตศาสตร์ความเร็วสูง
import matplotlib.pyplot as plt # นำเข้าไลบรารี Matplotlib สำหรับการสร้างพล็อตและการแสดงผลภาพบนหน้าจอ

# นำเข้ารูปภาพที่ต้องการ
img_bgr = cv2.imread("dog.jpg") # อ่านไฟล์ภาพ 'Dog.jpg' เข้ามาเก็บในตัวแปร img_bgr โดยปริยายจะอยู่ในโหมดสี BGR

# แปลงสีรูปจาก bgr ไปเป็น rgb
img_rgb = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2RGB) # แปลงสเปซสีจาก BGR เป็น RGB เนื่องจาก Matplotlib ใช้ระบบสี RGB ในการแสดงผล

# สกัดแชนเนลแต่ละสี
R = img_rgb[:, :, 0] # ดึงเฉพาะข้อมูลพิกเซลของแชนเนลสีแดง (Red) ซึ่งอยู่เลเยอร์ที่ 0 ของมิติที่ 3
G = img_rgb[:, :, 1] # ดึงเฉพาะข้อมูลพิกเซลของแชนเนลสีเขียว (Green) ซึ่งอยู่เลเยอร์ที่ 1 ของมิติที่ 3
B = img_rgb[:, :, 2] # ดึงเฉพาะข้อมูลพิกเซลของแชนเนลสีน้ำเงิน (Blue) ซึ่งอยู่เลเยอร์ที่ 2 ของมิติที่ 3

print(R, G, B) # พิมพ์ค่าอาร์เรย์ของแต่ละแชนเนลออกมาดูโครงสร้างพิกเซล

# สูตรมาตรฐาน ITU-R BT.601 / Luminance
# Gray = 0.299·R + 0.587·G + 0.114·B

# เนื่องจากค่าสีมีค่าเป็น int หากจะมาคำนวณกับสูตรได้ ต้องแปลงค่าเป็น float เพื่อกัน error ในการคูณทศนิยม
R_f = R.astype(np.float64) # แปลงชนิดข้อมูลแชนเนลสีแดงให้เป็นจุดทศนิยมขนาด 64 บิต
G_f = G.astype(np.float64) # แปลงชนิดข้อมูลแชนเนลสีเขียวให้เป็นจุดทศนิยมขนาด 64 บิต
B_f = B.astype(np.float64) # แปลงชนิดข้อมูลแชนเนลสีน้ำเงินให้เป็นจุดทศนิยมขนาด 64 บิต

# แทนค่าลงในสูตร
gray_manual = 0.299 * R_f + 0.587 * G_f + 0.114 * B_f # คำนวณความสว่างตามสูตร Photopic Curve ให้ได้ภาพ Grayscale

# แปลงค่าที่ได้จากการคำนวณเป็นชนิด int
gray_manual = np.clip(gray_manual, 0, 255).astype(np.uint8) # ตัดค่าผลลัพธ์ที่อาจเกินให้จำกัดอยู่ในช่วง [0, 255] และแปลงชนิดข้อมูลกลับเป็น uint8 (0-255)

# สร้างภาพ gray จาก opencv เพื่อนำมาเปรียบเทียบ
gray_cv2 = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2GRAY) # ใช้คำสั่งสำเร็จรูปของ OpenCV แปลงเป็น Grayscale โดยตรงเพื่อเทียบความแตกต่าง

fig, axes = plt.subplots(1, 2, figsize=(12, 5)) # สร้างรูปภาพหลักและแบ่งเป็น 2 กราฟย่อยแนวนอน ขนาดโดยรวม 12x5 นิ้ว
axes[0].imshow(gray_manual, cmap='gray') # พล็อตรูป Grayscale ที่คำนวณด้วยมือลงในกราฟแรก พร้อมระบุโทนสีเป็น 'gray'
axes[0].set_title('Manual Gray\n0.299R + 0.587G + 0.114B') # ตั้งชื่อหัวข้อให้กราฟย่อยแรก
axes[0].axis('off') # ปิดการแสดงผลเส้นแกนพิกัดและสเกลเพื่อความสวยงาม

axes[1].imshow(gray_cv2, cmap='gray') # พล็อตรูป Grayscale ที่ได้จาก OpenCV ลงในกราฟที่สอง
axes[1].set_title('OpenCV cvtColor\n(BGR2GRAY)') # ตั้งชื่อหัวข้อให้กราฟย่อยที่สอง
axes[1].axis('off') # ปิดการแสดงผลเส้นแกนพิกัด

plt.suptitle('Grayscale Comparison', fontsize=14) # เพิ่มชื่อหัวเรื่องหลักกึ่งกลางด้านบนสุดของรูป
plt.tight_layout() # จัดระเบียบระยะห่างระหว่างกราฟย่อยให้สมดุลและไม่ทับซ้อนกัน
plt.savefig('comparison.png', dpi=150) # บันทึกรูปกราฟลงในเครื่องด้วยชื่อ 'comparison.png' ที่ความละเอียด 150 DPI
plt.show() # แสดงผลกราฟทั้งหมดขึ้นมาบนหน้าจอ

# บันทึกไฟล์ผลลัพธ์
cv2.imwrite('my_gray_output.png', gray_manual) # ส่งออกอาร์เรย์ gray_manual บันทึกเป็นไฟล์รูปภาพ
print("บันทึก my_gray_output.png เรียบร้อยแล้ว") # แจ้งข้อความยืนยันการเซฟไฟล์ทางหน้าคอนโซล

คำถามท้ายการทดลองข้อที่ 1¶

อภิปรายว่าเมื่อนำค่าความสว่างที่แปลงด้วยสูตร 0.299·R + 0.587·G + 0.114·B ไปเทียบกับ ฟังก์ชันสำเร็จรูปของ OpenCV แล้ว ผลลัพธ์มีความแตกต่างกันในระดับสายตาและในเชิงค่าตัวเลขข้อมูลพิกเซลหรือไม่ย่างไร?

ตอบ: ไม่เห็นความแตกต่างระดับสายตา แต่ละระดับเชิงค่าตัวเลขมีต่างกันเล็กน้อย +- ไม่เกิน 1-2 พิกเซล

# โจทย์ข้อที่ 2
# การคัดเลือกและสกัดพื้นที่เฉพาะจุด (Region of Interest - ROI Extraction)

# นำเข้ารูปภาพที่ต้องการ
img_bgr = cv2.imread("dog.jpg") # โหลดภาพต้นฉบับเข้ามาใหม่ในโหมด BGR

# อ่าน shape เพื่อนำมาคำนวณหาใจกลางภาพ
h, w, c = img_bgr.shape # แยกข้อมูลขนาดอาร์เรย์ออกเป็น h (ความสูง), w (ความกว้าง), c (จำนวนช่องสี)
print(f"ขนาดภาพ: {w}x{h} pixels, {c} channels") # แสดงผลขนาดภาพดั้งเดิม

# กำหนดค่า roi ตามโจทย์
roi_w = 200 # ตั้งค่าเป้าหมายความกว้างของขอบเขต ROI ที่ต้องการตัด (200 พิกเซล)
roi_h = 200 # ตั้งค่าเป้าหมายความสูงของขอบเขต ROI ที่ต้องการตัด (200 พิกเซล)

# คำนวณพิกัดมุมบนซ้าย (x1, y1) และมุมล่างขวา (x2, y2)
x1 = (w - roi_w) // 2 # คำนวณพิกัดแกน X เริ่มต้น โดยหาพื้นที่เหลือจากความกว้างแล้วหารครึ่ง (ให้อยู่กึ่งกลาง)
y1 = (h - roi_h) // 2 # คำนวณพิกัดแกน Y เริ่มต้น โดยหาพื้นที่เหลือจากความสูงแล้วหารครึ่ง (ให้อยู่กึ่งกลาง)
x2 = x1 + roi_w # คำนวณพิกัดแกน X สิ้นสุด ด้วยการบวกความกว้างที่ต้องการเข้าไป
y2 = y1 + roi_h # คำนวณพิกัดแกน Y สิ้นสุด ด้วยการบวกความสูงที่ต้องการเข้าไป

print(f"พิกัด ROI: ({x1}, {y1}) -> ({x2}, {y2})") # แสดงผลพิกัดซ้ายบนและขวาล่างที่จะถูกตัด

# ทำการสไลซ์ (slice) อาเรย์เพื่อตัดภาพ (y1:y2 เลือกแถวจากบนลงล่าง, x1:x2 เลือกคอลัมน์จากซ้ายไปขวา)
roi = img_bgr[y1:y2, x1:x2] # สกัดข้อมูลส่วน ROI ตามพิกัดที่คำนวณไว้มาสร้างเป็นอาร์เรย์ภาพใหม่
print(f"ขนาด ROI ที่ได้: {roi.shape}") # ตรวจสอบขนาดอาร์เรย์ที่ตัดออกมาว่าตรงตามที่ต้องการหรือไม่ (200x200x3)

# เตรียมวาดกรอบสีแดงทับลงรูปเดิม
img_marked = img_bgr.copy() # ทำการสำเนาภาพเพื่อป้องกันไม่ให้ตอนที่วาดเส้นสี่เหลี่ยมไปกระทบตัวแปรภาพดั้งเดิม

# สร้างกรอบสีแดง
cv2.rectangle( # เรียกใช้คำสั่งวาดรูปสี่เหลี่ยมบนอาร์เรย์ภาพ
    img_marked, # ระบุตัวแปรอาร์เรย์ภาพเป้าหมายที่จะวาด
    (x1, y1), # กำหนดพิกัดมุมซ้ายบนของกรอบ
    (x2, y2), # กำหนดพิกัดมุมขวาล่างของกรอบ
    (0, 0, 255), # กำหนดสีของกรอบ (ระบบ BGR 0,0,255 คือสีแดง)
    3 # กำหนดความหนาของเส้นกรอบเท่ากับ 3 พิกเซล
)

# บันทึกไฟล์
cv2.imwrite('roi_center.jpg', roi) # ส่งออกและบันทึกภาพ ROI ที่ครอปมาได้เป็นไฟล์รูปภาพ
cv2.imwrite('sample_marked.jpg', img_marked) # ส่งออกและบันทึกภาพที่วาดกรอบสีแดงแล้ว
print("บันทึกไฟล์เรียบร้อย: roi_center.jpg, sample_marked.jpg") # แจ้งข้อความยืนยันการเซฟไฟล์สำเร็จ

# แสดงผลเปรียบเทียบ
fig, axes = plt.subplots(1, 2, figsize=(12, 5)) # สร้างกราฟแสดงผลคู่กันแนวนอน 2 รูป
axes[0].imshow(cv2.cvtColor(img_marked, cv2.COLOR_BGR2RGB)) # นำภาพวาดกรอบแปลงเป็น RGB และพล็อตลงช่องซ้าย
axes[0].set_title(f'Original + Red Rectangle\n({w}x{h})') # ตั้งชื่อภาพพร้อมระบุขนาดรวม
axes[0].axis('off') # ปิดสเกลพิกัด

axes[1].imshow(cv2.cvtColor(roi, cv2.COLOR_BGR2RGB)) # นำภาพส่วน ROI แปลงเป็น RGB และพล็อตลงช่องขวา
axes[1].set_title(f'ROI Crop\n({roi_w}x{roi_h})') # ตั้งชื่อภาพพร้อมระบุขนาดที่ตัด
axes[1].axis('off') # ปิดสเกลพิกัด

plt.tight_layout() # ปรับระเบียบระยะช่องว่างของภาพ
plt.show() # ดำเนินการแสดงกราฟหน้าจอ

ขนาดภาพ: 1400x1815 pixels, 3 channels
พิกัด ROI: (600, 807) -> (800, 1007)
ขนาด ROI ที่ได้: (200, 200, 3)
บันทึกไฟล์เรียบร้อย: roi_center.jpg, sample_marked.jpg

คำถามท้ายการทดลองข้อที่ 2¶

อธิบายว่าการสับเปลี่ยนสัญกรณ์แชนเนลสีจาก BGR เป็น RGB ส่งผลกระทบอย่างไรต่องานประเภทสกัดวัตถุที่มีสีเฉพาะเจาะจง?

ตอบ: การสับเปลี่ยนสัญกรณ์แชนเนลสีจาก BGR เป็น RGB ส่งผลกระทบต่องานสกัดวัตถุที่มีสีเฉพาะเจาะจงอย่างมีนัยสำคัญ เนื่องจาก OpenCV อ่านภาพในรูปแบบ BGR ตามค่าเริ่มต้น หากผู้อ้างอิง index แชนเนลโดยไม่แปลงรูปแบบก่อน จะทำให้การ threshold ผิดแชนเนล
